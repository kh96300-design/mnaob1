/** * دالة التشغيل الرئيسية لعرض الواجهة المستخدم (HTML) عند فتح التطبيق. */ function doGet() { return HtmlService.createHtmlOutputFromFile('Index') .setTitle('سجل المناوبة المطور - مدرسة الشيخ جاعد') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } /** * جلب بيانات جدول المناوبة بالكامل لإرسالها إلى واجهة الويب. */ function getScheduleData() { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("جدول المناوبة"); return sheet ? sheet.getDataRange().getDisplayValues() : []; } catch (e) { return []; } } /** * توليد جدول المناوبة لمدة 6 أشهر بناءً على "يوم المناوبة" المحدد لكل معلم في شيت المعلمين. */ function generateDutySchedule() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var tSheet = ss.getSheetByName("المعلمين"); if (!tSheet) { SpreadsheetApp.getUi().alert("خطأ: لم يتم العثور على ورقة باسم 'المعلمين'!"); return; } var tData = tSheet.getDataRange().getValues(); if (tData.length <= 1) { SpreadsheetApp.getUi().alert("خطأ: ورقة المعلمين فارغة أو تحتوي على العناوين فقط."); return; } // تنظيف العناوين والبحث عن الأعمدة المطلوبة var headers = tData[0].map(function(h) { return h.toString().trim(); }); var nameIdx = headers.indexOf("الاسم"); var dayIdx = headers.indexOf("يوم المناوبة"); // العمود الجديد المستهدف var phoneIdx = headers.indexOf("الهاتف"); var subIdx = headers.indexOf("التخصص"); // للاحتفاظ به للعرض فقط إن وجد if (nameIdx === -1 || dayIdx === -1) { SpreadsheetApp.getUi().alert("خطأ: تأكد من وجود أعمدة باسم 'الاسم' و 'يوم المناوبة' في ورقة المعلمين."); return; } // تصنيف المعلمين بناءً على الأيام المحددة لهم // الأحد (0)، الإثنين (1)، الثلاثاء (2)، الأربعاء (3)، الخميس (4) var teachersByDay = { 0: [], 1: [], 2: [], 3: [], 4: [] }; for (var i = 1; i < tData.length; i++) { var name = tData[i][nameIdx]; var dutyDay = tData[i][dayIdx]; var phone = phoneIdx !== -1 ? tData[i][phoneIdx] : ""; var subject = subIdx !== -1 ? tData[i][subIdx] : "-"; if (name && dutyDay) { var dayStr = dutyDay.toString().trim(); var targetDayNum = -1; // تحويل اسم اليوم المكتوب إلى رقم يتوافق مع نظام التواريخ (Date.getDay) if (dayStr.indexOf("أحد") !== -1 || dayStr.indexOf("الاحد") !== -1) targetDayNum = 0; else if (dayStr.indexOf("اثنين") !== -1 || dayStr.indexOf("الإثنين") !== -1) targetDayNum = 1; else if (dayStr.indexOf("ثلاثاء") !== -1 || dayStr.indexOf("الثلاثاء") !== -1) targetDayNum = 2; else if (dayStr.indexOf("أربعاء") !== -1 || dayStr.indexOf("الأربعاء") !== -1) targetDayNum = 3; else if (dayStr.indexOf("خميس") !== -1 || dayStr.indexOf("الخميس") !== -1) targetDayNum = 4; if (targetDayNum !== -1) { teachersByDay[targetDayNum].push({ name: name.toString().trim(), subject: subject.toString().trim(), phone: phone.toString().trim() }); } } } // إعداد مؤشرات تدوير مستقلة لكل يوم (في حال وجود أكثر من معلم في نفس اليوم) var rotationIndexes = { 0: 0, 1: 0, 2: 0, 3: 0, 4: 0 }; var dSheet = ss.getSheetByName("جدول المناوبة") || ss.insertSheet("جدول المناوبة"); dSheet.clear(); var finalRows = [["اليوم والتاريخ", "التخصص", "المعلم المناوب", "الهاتف"]]; // إعداد نطاق الـ 6 أشهر (من اليوم الحالي) var startDate = new Date(); var endDate = new Date(); endDate.setMonth(endDate.getMonth() + 6); var curr = new Date(startDate.getTime()); // حلقة تكرار تمر على الأيام يوماً تلو الآخر while (curr <= endDate) { var dayOfWeek = curr.getDay(); // استثناء الإجازة الأسبوعية (الجمعة 5 والسبت 6) if (dayOfWeek === 5 || dayOfWeek === 6) { curr.setDate(curr.getDate() + 1); continue; } var teachersForThisDay = teachersByDay[dayOfWeek]; var currentTeacher = null; // التحقق من وجود معلمين مخصصين لهذا اليوم if (teachersForThisDay && teachersForThisDay.length > 0) { var currentIdx = rotationIndexes[dayOfWeek]; currentTeacher = teachersForThisDay[currentIdx]; // التدوير بين المعلمين المخصصين لنفس هذا اليوم للأسبوع القادم rotationIndexes[dayOfWeek] = (currentIdx + 1) % teachersForThisDay.length; } else { currentTeacher = { name: "لا يوجد معلم مخصص لهذا اليوم", subject: "-", phone: "-" }; } var dayName = ["الأحد", "الإثنين", "الثلاثاء", "الأربعاء", "الخميس", "الجمعة", "السبت"][dayOfWeek]; var dateStr = dayName + " " + curr.getDate() + "/" + (curr.getMonth() + 1) + "/" + curr.getFullYear(); finalRows.push([ dateStr, currentTeacher.subject, currentTeacher.name, currentTeacher.phone ]); // الانتقال لليوم التالي curr.setDate(curr.getDate() + 1); } // كتابة البيانات وتنسيقها في الشيت dSheet.getRange(1, 1, finalRows.length, 4).setValues(finalRows); // ضبط الاتجاه والتنسيق الجمالي dSheet.setRightToLeft(true); var headerRange = dSheet.getRange(1, 1, 1, 4); headerRange.setFontWeight("bold") .setBackground("#1e3a8a") .setFontColor("white") .setHorizontalAlignment("center"); dSheet.getRange(2, 1, finalRows.length - 1, 4).setHorizontalAlignment("center"); dSheet.autoResizeColumns(1, 4); SpreadsheetApp.getUi().alert("تم بنجاح! تم توليد جدول الـ 6 أشهر بناءً على عمود 'يوم المناوبة' الثابت لكل معلم."); } /** * توثيق وحفظ عمليات تبادل المناوبات المكتملة في شيت منفصل يدعى "التبادل". */ function saveSwapToSheet(swapData) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName("التبادل"); if (!sheet) { sheet = ss.insertSheet("التبادل"); sheet.setRightToLeft(true); sheet.appendRow(["تاريخ تسجيل العملية", "اسم المعلم الأصلي", "تاريخ المناوبة المستبدلة", "تم التبديل مع المعلم"]); sheet.getRange("A1:D1").setFontWeight("bold").setBackground("#f1f5f9"); } const timestamp = Utilities.formatDate(new Date(), ss.getSpreadsheetTimeZone(), "yyyy/MM/dd HH:mm:ss"); sheet.appendRow([timestamp, swapData.teacherName, swapData.dutyDate, swapData.swappedWith]); return true; }
