} /** * جلب بيانات جدول المناوبة بالكامل لإرسالها إلى واجهة الويب. */ function getScheduleData() { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("جدول المناوبة"); return sheet ? sheet.getDataRange().getDisplayValues() : []; } catch (e) { return []; } } /** * توليد جدول المناوبة لمدة 6 أشهر بناءً على "يوم المناوبة" المحدد لكل معلم في شيت المعلمين. */ function generateDutySchedule() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var tSheet = ss.getSheetByName("المعلمين"); if (!tSheet) { SpreadsheetApp.getUi().alert("خطأ: لم يتم العثور على ورقة باسم 'المعلمين'!"); return; } var tData = tSheet.getDataRange().getValues(); if (tData.length <= 1) { SpreadsheetApp.getUi().alert("خطأ: ورقة المعلمين فارغة أو تحتوي على العناوين فقط."); return; } var headers = tData[0].map(function(h) { return h.toString().trim(); }); var nameIdx = headers.indexOf("الاسم"); var dayIdx = headers.indexOf("يوم المناوبة"); var phoneIdx = headers.indexOf("الهاتف"); var subIdx = headers.indexOf("التخصص"); if (nameIdx === -1 || dayIdx === -1) { SpreadsheetApp.getUi().alert("خطأ: تأكد من وجود أعمدة باسم 'الاسم' و 'يوم المناوبة' في ورقة المعلمين."); return; } var teachersByDay = { 0: [], 1: [], 2: [], 3: [], 4: [] }; for (var i = 1; i < tData.length; i++) { var name = tData[i][nameIdx]; var dutyDay = tData[i][dayIdx]; var phone = phoneIdx !== -1 ? tData[i][phoneIdx] : ""; var subject = subIdx !== -1 ? tData[i][subIdx] : "-"; if (name && dutyDay) { var dayStr = dutyDay.toString().trim(); var targetDayNum = -1; if (dayStr.indexOf("أحد") !== -1 || dayStr.indexOf("الاحد") !== -1) targetDayNum = 0; else if (dayStr.indexOf("اثنين") !== -1 || dayStr.indexOf("الإثنين") !== -1) targetDayNum = 1; else if (dayStr.indexOf("ثلاثاء") !== -1 || dayStr.indexOf("الثلاثاء") !== -1) targetDayNum = 2; else if (dayStr.indexOf("أربعاء") !== -1 || dayStr.indexOf("الأربعاء") !== -1) targetDayNum = 3; else if (dayStr.indexOf("خميس") !== -1 || dayStr.indexOf("الخميس") !== -1) targetDayNum = 4; if (targetDayNum !== -1) { teachersByDay[targetDayNum].push({ name: name.toString().trim(), subject: subject.toString().trim(), phone: phone.toString().trim() }); } } } var rotationIndexes = { 0: 0, 1: 0, 2: 0, 3: 0, 4: 0 }; var dSheet = ss.getSheetByName("جدول المناوبة") || ss.insertSheet("جدول المناوبة"); dSheet.clear(); var finalRows = [["اليوم والتاريخ", "التخصص", "المعلم المناوب", "الهاتف"]]; var startDate = new Date(); var endDate = new Date(); endDate.setMonth(endDate.getMonth() + 6); var curr = new Date(startDate.getTime()); while (curr <= endDate) { var dayOfWeek = curr.getDay(); if (dayOfWeek === 5 || dayOfWeek === 6) { curr.setDate(curr.getDate() + 1); continue; } var teachersForThisDay = teachersByDay[dayOfWeek]; var currentTeacher = null; if (teachersForThisDay && teachersForThisDay.length > 0) { var currentIdx = rotationIndexes[dayOfWeek]; currentTeacher = teachersForThisDay[currentIdx]; rotationIndexes[dayOfWeek] = (currentIdx + 1) % teachersForThisDay.length; } else { currentTeacher = { name: "لا يوجد معلم مخصص لهذا اليوم", subject: "-", phone: "-" }; } var dayName = ["الأحد", "الإثنين", "الثلاثاء", "الأربعاء", "الخميس", "الجمعة", "السبت"][dayOfWeek]; var dateStr = dayName + " " + curr.getDate() + "/" + (curr.getMonth() + 1) + "/" + curr.getFullYear(); finalRows.push([ dateStr, currentTeacher.subject, currentTeacher.name, currentTeacher.phone ]); curr.setDate(curr.getDate() + 1); } dSheet.getRange(1, 1, finalRows.length, 4).setValues(finalRows); dSheet.setRightToLeft(true); var headerRange = dSheet.getRange(1, 1, 1, 4); headerRange.setFontWeight("bold").setBackground("#1e3a8a").setFontColor("white").setHorizontalAlignment("center"); dSheet.getRange(2, 1, finalRows.length - 1, 4).setHorizontalAlignment("center"); dSheet.autoResizeColumns(1, 4); SpreadsheetApp.getUi().alert("تم بنجاح! تم توليد الجدول."); } /** * توثيق وحفظ عمليات تبادل المناوبات بناءً على الرمز السري الموحد (123). * @param {Object} swapData يحتوي على بيانات التبادل بالإضافة إلى الرقم السري المدخل. */ function saveSwapToSheet(swapData) { const ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. التحقق الفوري من رمز الدخول الموحد (123) const REQUIRED_KEY = "123"; if (!swapData.secretKey || swapData.secretKey.toString().trim() !== REQUIRED_KEY) { throw new Error("عذراً، رمز الدخول المدخل غير صحيح! يرجى إدخال الرمز الصحيح لتفعيل التبادل."); } // 2. التحقق من وجود المعلم في ورقة "المعلمين" let tSheet = ss.getSheetByName("المعلمين"); if (!tSheet) { throw new Error("خطأ نظام: لم يتم العثور على ورقة 'المعلمين'."); } let tData = tSheet.getDataRange().getValues(); let headers = tData[0].map(function(h) { return h.toString().trim(); }); let nameIdx = headers.indexOf("الاسم"); if (nameIdx === -1) { throw new Error("خطأ تهيئة: لم يتم العثور على عمود 'الاسم' في شيت المعلمين."); } let teacherExists = false; for (let i = 1; i < tData.length; i++) { if (tData[i][nameIdx].toString().trim() === swapData.teacherName.trim()) { teacherExists = true; break; } } if (!teacherExists) { throw new Error("تنبيه: اسم المعلم الأصلي غير مسجل في قائمة 'المعلمين'."); } // 3. التوثيق والتسجيل في شيت "التبادل" let sheet = ss.getSheetByName("التبادل"); if (!sheet) { sheet = ss.insertSheet("التبادل"); sheet.setRightToLeft(true); sheet.appendRow(["تاريخ تسجيل العملية", "اسم المعلم الأصلي", "تاريخ المناوبة المستبدلة", "تم التبديل مع المعلم", "حالة العملية"]); sheet.getRange("A1:E1").setFontWeight("bold").setBackground("#1e3a8a").setFontColor("white"); } const timestamp = Utilities.formatDate(new Date(), ss.getSpreadsheetTimeZone(), "yyyy/MM/dd HH:mm:ss"); sheet.appendRow([ timestamp, swapData.teacherName, swapData.dutyDate, swapData.swappedWith, "تم الاعتماد برمز الدخول الموحد 123" ]); sheet.autoResizeColumns(1, 5); return true; }
